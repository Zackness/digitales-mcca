static void fontGetGlyphColumns(char c, uint8_t outCols5[5]) {

  if (c >= 'a' && c <= 'z') c = (char)(c - 'a' + 'A');

  // Por defecto: "?"

  outCols5[0] = qmark[0];
  outCols5[1] = qmark[1];
  outCols5[2] = qmark[2];
  outCols5[3] = qmark[3];
  outCols5[4] = qmark[4];

  switch (c) {
    case ' ':
      outCols5[0] = 0x00; outCols5[1] = 0x00; outCols5[2] = 0x00; outCols5[3] = 0x00; outCols5[4] = 0x00;
      break;

      outCols5[0] = 0x08; outCols5[1] = 0x08; outCols5[2] = 0x08; outCols5[3] = 0x08; outCols5[4] = 0x08;

// =============================
// MAX7219 Low-Level (SPI nativo)
// =============================
static inline void maxCSLow() { digitalWrite(PIN_CS, LOW); }
static inline void maxCSHigh() { digitalWrite(PIN_CS, HIGH); }

static void maxSendBroadcast(uint8_t reg, uint8_t data) {
  // En cascada: los datos del "último chip" deben ir primero en el shift.
  maxCSLow();
  for (int chip = (int)NUM_CHIPS - 1; chip >= 0; --chip) {
    maxSpi.transfer(reg);
    maxSpi.transfer(data);
  }
  maxCSHigh();
}

static void maxSendRow(uint8_t row, const uint8_t rowBytes[NUM_CHIPS]) {
  // row: 0..7 => registro DIGIT1..DIGIT8 => reg = row+1
  maxCSLow();
  for (int chip = (int)NUM_CHIPS - 1; chip >= 0; --chip) {
    maxSpi.transfer((uint8_t)(row + 1));
    maxSpi.transfer(rowBytes[chip]);
  }
  maxCSHigh();
}

static void setIntensityAll(uint8_t intensity) {
  gIntensity = intensity & 0x0F;
  maxSendBroadcast(0x0A, gIntensity); // INTENSITY reg
}

static void maxInit() {
  pinMode(PIN_CS, OUTPUT);
  maxCSHigh();

  maxSpi.begin(PIN_CLK, PIN_MISO, PIN_DIN, PIN_CS);

  maxSendBroadcast(0x0F, 0x00); // display test OFF
  maxSendBroadcast(0x09, 0x00); // decode OFF
  maxSendBroadcast(0x0B, 0x07); // scan limit 8 filas
  maxSendBroadcast(0x0A, gIntensity & 0x0F); // intensidad
  maxSendBroadcast(0x0C, 0x01); // ENCENDER
}

static void clearBuffer() {
  for (uint8_t y = 0; y < ROWS; ++y) rowMasks[y] = 0;
  hwDirty = true;
}

static inline void setPixel(uint8_t x, uint8_t y, bool on) {
  if (x >= COLS || y >= ROWS) return;
  if (on) rowMasks[y] |= (1UL << x);
  else rowMasks[y] &= ~(1UL << x);
}

static void updateHardware() {
  // Renderiza buffer->MAX7219
  if (!hwDirty) return;
  for (uint8_t y = 0; y < ROWS; ++y) {
    uint8_t bytes[NUM_CHIPS];
    for (uint8_t chip = 0; chip < NUM_CHIPS; ++chip) {
      uint8_t b = 0;
      for (uint8_t xInChip = 0; xInChip < 8; ++xInChip) {
        uint8_t x = (uint8_t)(chip * 8 + xInChip);
        bool on = (rowMasks[y] & (1UL << x)) != 0;

        // MAX7219: típicamente el bit 7 corresponde al "lado izquierdo" del módulo 8x8.
        if (on) b |= (1U << (7 - xInChip));
      }
      // bytes invertido por cascada (para 8x32)
      bytes[NUM_CHIPS - 1 - chip] = b;
    }
    maxSendRow(y, bytes);
  }
  hwDirty = false;
}

// =============================
// Modo texto (scroll)
// =============================
static constexpr uint16_t MAX_TEXT_LEN = 64;
static constexpr uint16_t CHAR_W = 5;
static constexpr uint16_t CHAR_SPACING = 1;
static constexpr uint16_t OFF_WIDTH_MAX = MAX_TEXT_LEN * (CHAR_W + CHAR_SPACING); // 384

static uint8_t offCols[OFF_WIDTH_MAX]; // cada columna guarda bits de filas (bit0..bit6)
static uint16_t offWidth = 0;
static int16_t scrollPos = 0;
static uint16_t scrollSpeedMs = 60;
static unsigned long lastScrollMs = 0;
static String currentText = "";
static bool currentInvert = false;

static void buildTextOffscreen(const String &text) {
  clearBuffer();

  String t = text;
  if (t.length() > MAX_TEXT_LEN) t = t.substring(0, MAX_TEXT_LEN);

  offWidth = (uint16_t)(t.length() * (CHAR_W + CHAR_SPACING));
  if (offWidth > OFF_WIDTH_MAX) offWidth = OFF_WIDTH_MAX;

  // Relleno: por cada caracter (5 columnas) + 1 columna de espacio
  uint16_t x = 0;
  for (size_t i = 0; i < t.length() && x < offWidth; ++i) {
    char c = t[i];
    uint8_t glyph[5];
    fontGetGlyphColumns(c, glyph);

    for (uint8_t col = 0; col < CHAR_W && x < offWidth; ++col) {
      offCols[x++] = glyph[col];
    }
    // Espacio
    if (x < offWidth) offCols[x++] = 0x00;
  }

  // Para que entre "desde la derecha"
  scrollPos = -((int16_t)COLS);
}

static void drawScrollFrame() {
  clearBuffer();

  for (uint8_t dX = 0; dX < COLS; ++dX) {
    int16_t globalX = (int16_t)scrollPos + (int16_t)dX;
    if (globalX < 0 || globalX >= (int16_t)offWidth) continue;

    uint8_t colMask = offCols[globalX];
    for (uint8_t y = 0; y < ROWS; ++y) {
      if (colMask & (1U << y)) {
        setPixel(dX, y, true);
      }
    }
  }
}

static void applyScrollText(const String &text, uint32_t speedMs) {
  uint32_t speed = speedMs;
  if (speed < 5) speed = 5;
  if (speed > 2000) speed = 2000;
  scrollSpeedMs = (uint16_t)speed;

  currentText = text;
  if (currentText.length() > MAX_TEXT_LEN) currentText = currentText.substring(0, MAX_TEXT_LEN);
  buildTextOffscreen(currentText);
  mode = DisplayMode::SCROLL_TEXT;
  hwDirty = true;
  drawScrollFrame();
  updateHardware();
}

  server.sendHeader("Access-Control-Allow-Origin", "*");
  server.sendHeader("Access-Control-Allow-Methods", "POST, GET, OPTIONS");
  server.sendHeader("Access-Control-Allow-Headers", "Content-Type");
  server.sendHeader("Access-Control-Max-Age", "86400");

              "POST /api/text {\"text\":\"HOLA\",\"speedMs\":60,\"intensity\":8}\n"
              "POST /api/pattern {\"bitmap\":[row0,row1,...row7]}\n"
              "POST /api/off\n"
              "GET  /api/state");
}

static void handleOff() {

  server.sendHeader("Access-Control-Allow-Origin", "*");
  server.sendHeader("Access-Control-Allow-Methods", "POST, GET, OPTIONS");
  server.sendHeader("Access-Control-Allow-Headers", "Content-Type");
  server.sendHeader("Access-Control-Max-Age", "86400");
  server.send(200, "application/json", "{\"ok\":true}");
}

static void handlePattern() {
  if (!server.hasArg("plain")) {
    server.sendHeader("Access-Control-Allow-Origin", "*");
    server.sendHeader("Access-Control-Allow-Methods", "POST, GET, OPTIONS");
    server.sendHeader("Access-Control-Allow-Headers", "Content-Type");
    server.sendHeader("Access-Control-Max-Age", "86400");
    server.send(400, "application/json", "{\"ok\":false,\"error\":\"Falta body JSON\"}");
    return;
  }

  String body = server.arg("plain");
  StaticJsonDocument<256> doc;
  DeserializationError err = deserializeJson(doc, body);
  if (err) {
    server.sendHeader("Access-Control-Allow-Origin", "*");
    server.sendHeader("Access-Control-Allow-Methods", "POST, GET, OPTIONS");
    server.sendHeader("Access-Control-Allow-Headers", "Content-Type");
    server.sendHeader("Access-Control-Max-Age", "86400");
    server.send(400, "application/json", "{\"ok\":false,\"error\":\"JSON invalido\"}");
    return;
  }

  JsonArray arr = doc["bitmap"].as<JsonArray>();
  if (arr.size() != ROWS) {
    server.sendHeader("Access-Control-Allow-Origin", "*");
    server.sendHeader("Access-Control-Allow-Methods", "POST, GET, OPTIONS");
    server.sendHeader("Access-Control-Allow-Headers", "Content-Type");
    server.sendHeader("Access-Control-Max-Age", "86400");
    server.send(400, "application/json", "{\"ok\":false,\"error\":\"bitmap debe tener 8 filas\"}");
    return;
  }

  bool invert = doc["invert"] | false;
  currentInvert = invert;
  for (uint8_t y = 0; y < ROWS; ++y) {
    uint32_t mask = (uint32_t)arr[y].as<uint32_t>();
    if (invert) mask = (~mask) & 0xFFFFFFFFUL;
    rowMasks[y] = mask;
  }

  if (doc.containsKey("intensity")) {
    gIntensity = ((uint8_t)doc["intensity"].as<uint32_t>()) & 0x0F;
    setIntensityAll(gIntensity);
  }

  mode = DisplayMode::STATIC_PATTERN;
  hwDirty = true;
  updateHardware();
  server.sendHeader("Access-Control-Allow-Origin", "*");
  server.sendHeader("Access-Control-Allow-Methods", "POST, GET, OPTIONS");
  server.sendHeader("Access-Control-Allow-Headers", "Content-Type");
  server.sendHeader("Access-Control-Max-Age", "86400");
  server.send(200, "application/json", "{\"ok\":true}");
}

    server.sendHeader("Access-Control-Allow-Headers", "Content-Type");
    server.sendHeader("Access-Control-Max-Age", "86400");
    server.send(400, "application/json", "{\"ok\":false,\"error\":\"Falta body JSON\"}");

  }

  String body = server.arg("plain");

  DeserializationError err = deserializeJson(doc, body);

    server.sendHeader("Access-Control-Allow-Origin", "*");

    server.sendHeader("Access-Control-Allow-Headers", "Content-Type");
    server.sendHeader("Access-Control-Max-Age", "86400");

  }

  applyScrollText(String(text), speed);

  server.sendHeader("Access-Control-Allow-Headers", "Content-Type");

static void handleOptions() {
  server.sendHeader("Access-Control-Allow-Origin", "*");
  server.sendHeader("Access-Control-Allow-Methods", "POST, GET, OPTIONS");
  server.sendHeader("Access-Control-Allow-Headers", "Content-Type");
  server.sendHeader("Access-Control-Max-Age", "86400");
  server.send(204);
}

static void handleState() {
  StaticJsonDocument<1024> doc;

  if (mode == DisplayMode::OFF) doc["mode"] = "off";
  else if (mode == DisplayMode::STATIC_PATTERN) doc["mode"] = "pattern";
  else if (mode == DisplayMode::SCROLL_TEXT) doc["mode"] = "text";

  doc["intensity"] = gIntensity;
  doc["speedMs"] = scrollSpeedMs;
  doc["invert"] = currentInvert;
  doc["text"] = currentText;

  JsonArray arr = doc.createNestedArray("bitmap");
  for (uint8_t y = 0; y < ROWS; ++y) arr.add(rowMasks[y]);

  String payload;
  serializeJson(doc, payload);

  server.sendHeader("Access-Control-Allow-Origin", "*");
  server.sendHeader("Access-Control-Allow-Methods", "POST, GET, OPTIONS");
  server.sendHeader("Access-Control-Allow-Headers", "Content-Type");
  server.sendHeader("Access-Control-Max-Age", "86400");
  server.send(200, "application/json", payload);
}

static void setupServer() {
  server.on("/", HTTP_GET, handleRoot);
  server.on("/api/off", HTTP_POST, handleOff);
  server.on("/api/pattern", HTTP_POST, handlePattern);
  server.on("/api/text", HTTP_POST, handleText);
  server.on("/api/state", HTTP_GET, handleState);

  // CORS preflight (OPTIONS)
  server.on("/api/off", HTTP_OPTIONS, handleOptions);
  server.on("/api/pattern", HTTP_OPTIONS, handleOptions);
  server.on("/api/text", HTTP_OPTIONS, handleOptions);
  server.on("/api/state", HTTP_OPTIONS, handleOptions);
  server.on("/", HTTP_OPTIONS, handleOptions);

  server.begin();
}

// Pruebas por monitor serie (USB)
// =============================
static String serialLineBuf;

static void processSerialInput() {
  while (Serial.available() > 0) {
    int r = Serial.read();
    if (r < 0) break;
    char c = (char)r;
    if (c == '\r') continue;
    if (c == '\n') {
      String line = serialLineBuf;
      serialLineBuf = "";
      line.trim();
      if (line.length() == 0) continue;

      if (line.equalsIgnoreCase("off")) {
        mode = DisplayMode::OFF;
        clearBuffer();
        updateHardware();
        Serial.println(F("[OK] Matriz apagada."));
        continue;
      }

      // speed <ms>
      if (line.length() > 6 && line.substring(0, 6).equalsIgnoreCase("speed ")) {
        int sp = line.substring(6).toInt();
        if (sp < 5) sp = 5;
        if (sp > 2000) sp = 2000;
        scrollSpeedMs = (uint16_t)sp;
        Serial.print(F("[OK] Velocidad scroll: "));
        Serial.print(scrollSpeedMs);
        Serial.println(F(" ms"));
        continue;
      }

      applyScrollText(line, scrollSpeedMs);
      Serial.print(F("[OK] Mostrando: "));
      Serial.println(line);
      continue;
    }
    if (serialLineBuf.length() < (int)MAX_TEXT_LEN + 16) {
      serialLineBuf += c;
    }
  }
}

static void connectWifiBlocking(uint32_t timeoutMs) {
  WiFi.mode(WIFI_STA);
  WiFi.setHostname(HOSTNAME);
  WiFi.begin(WIFI_SSID, WIFI_PASS);

  unsigned long start = millis();
  while (WiFi.status() != WL_CONNECTED) {
    delay(250);
    if (millis() - start > timeoutMs) break;
  }
}

static void ensureWifiConnected() {
  if (WiFi.status() == WL_CONNECTED) return;
  unsigned long now = millis();
  if (now - lastWifiRetryMs < WIFI_RETRY_EVERY_MS) return;
  lastWifiRetryMs = now;
  WiFi.disconnect(false);
  WiFi.begin(WIFI_SSID, WIFI_PASS);
}

// =============================
// CLOUD helpers (Vercel)
// =============================
static bool cloudPostJson(const String &path, const String &jsonBody, String &outResponse, int &outCode) {
  if (WiFi.status() != WL_CONNECTED) return false;

  WiFiClientSecure client;
  client.setInsecure();

  HTTPClient http;
  http.setTimeout(3500);

  String url = String(CLOUD_BASE_URL) + path;
  if (!http.begin(client, url)) return false;

  http.addHeader("Content-Type", "application/json");
  http.addHeader("x-device-token", DEVICE_TOKEN);

  outCode = http.POST((uint8_t *)jsonBody.c_str(), jsonBody.length());
  outResponse = (outCode > 0) ? http.getString() : "";
  http.end();
  return outCode > 0;
}

static void cloudPollAndApply() {
  if (WiFi.status() != WL_CONNECTED) return;
  unsigned long now = millis();
  if (now - lastPollMs < POLL_EVERY_MS) return;
  lastPollMs = now;

  StaticJsonDocument<96> req;
  req["deviceId"] = DEVICE_ID;
  String reqBody;
  serializeJson(req, reqBody);

  String resp;
  int code = 0;
  if (!cloudPostJson("/api/device/poll", reqBody, resp, code)) return;
  if (code != 200) return;

  StaticJsonDocument<768> doc;
  DeserializationError err = deserializeJson(doc, resp);
  if (err) return;

  if (!(doc["ok"] | false)) return;
  if (doc["command"].isNull()) return;

  const char *type = doc["command"]["type"] | "";

  if (strcmp(type, "off") == 0) {
    mode = DisplayMode::OFF;
    clearBuffer();
    updateHardware();
    return;
  }

  if (strcmp(type, "text") == 0) {
    const char *text = doc["command"]["text"] | "";
    uint32_t speedMs = doc["command"]["speedMs"] | scrollSpeedMs;

    if (doc["command"].containsKey("intensity")) {
      gIntensity = ((uint8_t)doc["command"]["intensity"].as<uint32_t>()) & 0x0F;
      setIntensityAll(gIntensity);
    }

    applyScrollText(String(text), speedMs);
    return;
  }

  if (strcmp(type, "pattern") == 0) {
    bool invert = doc["command"]["invert"] | false;
    currentInvert = invert;

    JsonArray arr = doc["command"]["bitmap"].as<JsonArray>();
    if (arr.size() != ROWS) return;

    for (uint8_t y = 0; y < ROWS; ++y) {
      uint32_t mask = (uint32_t)arr[y].as<uint32_t>();
      if (invert) mask = (~mask) & 0xFFFFFFFFUL;
      rowMasks[y] = mask;
    }

    if (doc["command"].containsKey("intensity")) {
      gIntensity = ((uint8_t)doc["command"]["intensity"].as<uint32_t>()) & 0x0F;
      setIntensityAll(gIntensity);
    }

    mode = DisplayMode::STATIC_PATTERN;
    hwDirty = true;
    updateHardware();
    return;
  }
}

static void cloudReportState() {
  if (WiFi.status() != WL_CONNECTED) return;
  unsigned long now = millis();
  if (now - lastReportMs < REPORT_EVERY_MS) return;
  lastReportMs = now;

  StaticJsonDocument<768> payload;
  payload["deviceId"] = DEVICE_ID;

  JsonObject state = payload.createNestedObject("state");
  if (mode == DisplayMode::OFF) state["mode"] = "off";
  else if (mode == DisplayMode::STATIC_PATTERN) state["mode"] = "pattern";
  else state["mode"] = "text";

  state["intensity"] = gIntensity;
  state["speedMs"] = scrollSpeedMs;
  state["invert"] = currentInvert;
  state["text"] = currentText;

  // bitmap (8x32)
  JsonArray arr = state.createNestedArray("bitmap");
  for (uint8_t y = 0; y < ROWS; ++y) arr.add(rowMasks[y]);

  String body;
  serializeJson(payload, body);

  String resp;
  int code = 0;
  (void)cloudPostJson("/api/device/report", body, resp, code);
}

void setup() {
  Serial.begin(115200);
  delay(100);

  connectWifiBlocking(20000);
  if (WiFi.status() == WL_CONNECTED) {
    Serial.print("IP: ");
    Serial.println(WiFi.localIP());
  } else {
    Serial.println("No se pudo conectar WiFi (ver SSID/PASS).");
  }

  Serial.println();
  Serial.println(F("=== Prueba por terminal (115200) ==="));
  Serial.println(F("Escribe texto y Enter -> scroll en la matriz (max 64 caracteres)."));
  Serial.println(F("Comandos: off | speed <5-2000>"));
  Serial.println();

  maxInit();
  clearBuffer();
  updateHardware();

  setupServer();
  mode = DisplayMode::OFF;
}

void loop() {
  server.handleClient();
  processSerialInput();

  ensureWifiConnected();

  // Control remoto por Vercel
  cloudPollAndApply();
  cloudReportState();

  // Scroll local
  if (mode == DisplayMode::SCROLL_TEXT) {
    unsigned long now = millis();
    if (now - lastScrollMs >= scrollSpeedMs) {
      lastScrollMs = now;
      drawScrollFrame();
      updateHardware();

      scrollPos++;
      if (scrollPos > (int16_t)offWidth) {
        scrollPos = -((int16_t)COLS);
      }
    }
  }
}
